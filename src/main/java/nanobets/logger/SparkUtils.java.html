1 package nanobets;
 2 //apache log4j
 3 import org.apache.log4j.Logger;
 4 //jetty
 5 import org.eclipse.jetty.server.AbstractNCSARequestLog;
 6 //spark
 7 import spark.embeddedserver.EmbeddedServers;
 8 import spark.embeddedserver.jetty.EmbeddedJettyFactory;
 9 
10 
11 public class SparkUtils {
12     public static void createServerWithRequestLog(Logger logger) {
13         EmbeddedJettyFactory factory = createEmbeddedJettyFactoryWithRequestLog(logger);
14         EmbeddedServers.add(EmbeddedServers.Identifiers.JETTY, factory);
15     }
16 
17     private static EmbeddedJettyFactory createEmbeddedJettyFactoryWithRequestLog(org.apache.log4j.Logger logger) {
18         AbstractNCSARequestLog requestLog = new RequestLogFactory(logger).create();
19         return new EmbeddedJettyFactoryConstructor(requestLog).create();
20     }

22 }
